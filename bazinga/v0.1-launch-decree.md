## src/[main.rs](http://main.rs) (Five Commands вЂ” Live)
```rust
use clap::{Parser, Subcommand};

#[derive(Parser)]
#[command(name = "bazinga", about = "One download. One constitution. Everything.", version = "0.1.0")]
struct Cli {
    #[command(subcommand)]
    command: Commands,
}

#[derive(Subcommand)]
enum Commands {
    SyncNotion,
    SyncDrive,
    SyncGithub,
    Index,
    Bench,
}

#[tokio::main]
async fn main() {
    let cli = Cli::parse();
    bazinga::engines::constitutional_engine::boot().await;
    match cli.command {
        Commands::SyncNotion => bazinga::commands::sync_notion::run().await,
        Commands::SyncDrive => bazinga::commands::sync_drive::run().await,
        Commands::SyncGithub => bazinga::commands::sync_github::run().await,
        Commands::Index => bazinga::commands::index::run().await,
        Commands::Bench => bazinga::commands::bench::run().await,
    }
}
```
